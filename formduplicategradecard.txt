  <script>
    function setupFileUpload(inputId, uploadDivId, fileNameId) {
      const input = document.getElementById(inputId);
      const uploadDiv = document.getElementById(uploadDivId);
      const fileNameDiv = document.getElementById(fileNameId);

      uploadDiv.addEventListener('click', (e) => {
        if (e.target !== input) {
          input.click();
        }
      });

      input.addEventListener('change', (e) => {
        if (e.target.files.length > 0) {
          const file = e.target.files[0];
          const fileSize = (file.size / (1024 * 1024)).toFixed(2);
          
          if (file.size > 1073741824) {
            alert('File size exceeds 1 GB limit!');
            input.value = '';
            fileNameDiv.innerHTML = '';
            return;
          }
          
          fileNameDiv.innerHTML = `<div>${file.name}</div><div class="file-size">${fileSize} MB</div>`;
        }
      });

      uploadDiv.addEventListener('dragover', (e) => {
        e.preventDefault();
        uploadDiv.classList.add('dragover');
      });

      uploadDiv.addEventListener('dragleave', () => {
        uploadDiv.classList.remove('dragover');
      });

      uploadDiv.addEventListener('drop', (e) => {
        e.preventDefault();
        uploadDiv.classList.remove('dragover');
        
        if (e.dataTransfer.files.length > 0) {
          input.files = e.dataTransfer.files;
          input.dispatchEvent(new Event('change'));
        }
      });
    }

    setupFileUpload('policeComplaint', 'policeComplaintUpload', 'policeComplaintName');
    setupFileUpload('affidavit', 'affidavitUpload', 'affidavitName');
    setupFileUpload('gradeCard', 'gradeCardUpload', 'gradeCardName');
    setupFileUpload('sbiReceipt', 'sbiReceiptUpload', 'sbiReceiptName');

    document.getElementById('duplicateGradeCardForm').addEventListener('submit', async (e) => {
      e.preventDefault();
      
      const submitBtn = document.getElementById('submitBtn');
      const form = document.getElementById('duplicateGradeCardForm');
      const loading = document.getElementById('loading');
      const errorMessage = document.getElementById('errorMessage');
      const successMessage = document.getElementById('successMessage');
      
      submitBtn.disabled = true;
      form.style.display = 'none';
      loading.style.display = 'block';
      errorMessage.style.display = 'none';
      
      try {
        const formData = {
          email: document.getElementById('email').value,
          applicantName: document.getElementById('applicantName').value,
          mobile: document.getElementById('mobile').value,
          regNo: document.getElementById('regNo').value,
          campus: document.getElementById('campus').value,
          program: document.getElementById('program').value,
          periodOfStudy: document.getElementById('periodOfStudy').value,
          semester: document.getElementById('semester').value,
          reason: document.getElementById('reason').value
        };
        
        const files = {};
        const fileInputs = ['policeComplaint', 'affidavit', 'gradeCard', 'sbiReceipt'];
        
        for (const inputId of fileInputs) {
          const input = document.getElementById(inputId);
          if (input.files.length > 0) {
            const file = input.files[0];
            const base64 = await readFileAsBase64(file);
            files[inputId] = {
              name: file.name,
              type: file.type,
              data: base64
            };
          }
        }
        
        google.script.run
          .withSuccessHandler((result) => {
            loading.style.display = 'none';
            successMessage.style.display = 'block';
            document.getElementById('appIdDisplay').textContent = result.applicationId;
          })
          .withFailureHandler((error) => {
            loading.style.display = 'none';
            form.style.display = 'block';
            errorMessage.textContent = 'Error: ' + error.message;
            errorMessage.style.display = 'block';
            submitBtn.disabled = false;
          })
          .submitDuplicateGradeCard(formData, files);
        
      } catch (error) {
        loading.style.display = 'none';
        form.style.display = 'block';
        errorMessage.textContent = 'Error: ' + error.message;
        errorMessage.style.display = 'block';
        submitBtn.disabled = false;
      }
    });
    
    function readFileAsBase64(file) {
      return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = () => resolve(reader.result);
        reader.onerror = reject;
        reader.readAsDataURL(file);
      });
    }
